Media queries for font size

$max-width => maximum ekran ölçüsü , $min-break => minimum ekran ölçüsü , $break => kaç adet font size olacağını belirler 

@mixin font-size-scale($max-width: 1366px, $min-break: 375px, $max: 6) {
	html {
		@for $i from 2 to $max {
			@media(max-width: $max-width - ceil(($max-width - $min-break)/($max - $i))) {
				font-size: 100% - ($i - 1)*6%;
			}
		}
	}
}

Font Set

$name => font name , $url => font url (string olarak girilmesi gerekiyor) , $type => font type (.otf) *font-weight'leri font-set map'in den çekmektedir. 

$font-set: ("light": 200, "regular": 400, "bold": 700, "black":900);
@mixin font-set($name,$url,$type) {
	@each $font-name,
	$font-weight in $font-set {
		@font-face {
			font-family: $name;
			src: url('#{$url}#{$font_weight}.#{$type}') format('opentype');
			font-weight: $font_weight;
			font-style: normal;
		}
	}
}

Font Face

$font-name => font ismi , $file-url => dosya yolu , $file-name => font dosyası ismi , $weight => font-weight varsayılan normal , $style => font-style varsayılan normal 

@mixin font-face($font-name,$file-url,$file-name,$weight: normal,$style: normal) {
	@font-face {
		font-family: quote($font-name);
		src: url($file-url + '/' + $file-name + '.eot');
		src: url($file-url + '/' + $file-name + '.eot?#iefix') format('embedded-opentype'),
		url($file-url + '/' + $file-name + '.woff') format('woff'),
		url($file-url + '/' + $file-name + '.ttf') format('truetype'),
		url($file-url + '/' + $file-name + '.svg##{$font-name}') format('svg');
		font-weight: $weight;
		font-style: $style;
	}
}

Default box shadow

$x => yatay , $y => dikey , $blur => bulanıklık değeri , $color => renk , $inset => inset durumu varsayılan olarak boş bir değer girilmedikçe normal box shadow olarak ekler inset boş olmazsa box shadow'a inset value ekler *box shadow'u prefixleyerek vermektedir. css3 prefix mixini içermektedir. 

@mixin box-shadow($x: 2px,$y: 2px,$blur: 5px,$color: #000,$inset: "") {
	@if ($inset !="") {
		@include css3-prefix('box-shadow',
		$inset $x $y $blur $color);
	}
	@else {
		@include css3-prefix('box-shadow',
		$x $y $blur $color);
	}
}


.example{
	@include box-shadow(2px,4px,5px,#e35252,inset)
}

.example{
	-webkit-box-shadow: inset 2px 4px 5px #e35252;
	-moz-box-shadow: inset 2px 4px 5px #e35252;
	-ms-box-shadow: inset 2px 4px 5px #e35252;
	-o-box-shadow: inset 2px 4px 5px #e35252;
	box-shadow: inset 2px 4px 5px #e35252;
}

Font Calculation

$weight => font-weight , $font-size => font-size , $line-height => line-height , $font-family => varsayılan değeri none herhangi birşey girerseniz kod'a ekleyecektir , $color => varsayılan transparenttir , $opacity varsayılan 1 dir girdiğiniz değere göre hesaplama yapar 

@mixin fontCalculation($weight,
	$font-size,
	$line-height,
	$font-family:none,
	$color:transparent,
	$opacity:1) {
		font: $weight $font-size/$globalRem*1rem #{'/'} $line-height/$globalRem*1rem;
		@if ($font-family != none){
			font-family:$font-family;
			
		}
		
		@if ($color != transparent){
			@include colorOpacity($color,$opacity);

		}
	
}
	


.example{
    @include fontCalculation(400,2,4,Nunito,#000,0.4)
}

.example{
	font: 400 0.125rem / 0.25rem;
	font-family: Nunito;
	color: rgba(0,0,0,0.4);
}

Arrow

$direction: üçgen yönü default olarak down , diğer yönler right,left,up , $size => üçgen boyutu , $color => üçgen rengi 

@mixin arrow($direction: down,
	$size: 5px,
	$color: #555) {
		width: 0;
		height: 0;
		@if ($direction==left) {
			border-top: $size solid transparent;
			border-bottom: $size solid transparent;
			border-right: $size solid $color;
		}
		@else if ($direction==right) {
			border-top: $size solid transparent;
			border-bottom: $size solid transparent;
			border-left: $size solid $color;
		}
		@else if ($direction==down) {
			border-left: $size solid transparent;
			border-right: $size solid transparent;
			border-top: $size solid $color;
		}
		@else {
			border-left: $size solid transparent;
			border-right: $size solid transparent;
			border-bottom: $size solid $color;
		}
}

}

Random Color Function

$r => red , $g => green , $b => blue , $a => alpha(opacity) 


@function colorRandom($r,$g,$b,$a){
	$red:null;
	$green:null;
	$blue:null;
	
	@if ($r >255){
		$red:random($limit:255);
	} @else {
		$red:random($r);
	}
	
	@if ($g >255){
		$green:random($limit:255);
	} @else {
		$green:random($g);
	}
	
	@if ($b > 255){
		$blue:random($limit:255);
	} @else {
		$blue:random($b);
	}
	
	
	@return grba($red,$green,$blue,$a);

	
}
	
	



.element{
	color:colorRandom(900,222,78,1);
}
	
.element {
  color: grba(246, 41, 28, 1);
}

@mixin colorPalette($color){

		}
		
		@for $i from 1 through 4{
		--color-lighten-#{$i * 10}: lighten($color,$i * 10deg);
		}
		
		@for $i from 1 through 4{
		--color-adjust-hue-#{$i * 10}: adjust-hue($color,$i * -10deg);
		}
		
		@for $i from 1 through 4{
		--color-adjust-minus-hue-#{$i * 10}: adjust-hue($color,$i * 10deg);
		}
		
		@for $i from 1 through 4{
		--color-saturate-#{$i * 10}: saturate($color,$i * 10deg);
