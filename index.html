Css3 Prefix

$prop => property (box-sizing) , $value => (border-box) 

@mixin css3-prefix($prop, $value) {
	-webkit-#{$prop}: #{$value};
	-moz-#{$prop}: #{$value};
	-ms-#{$prop}: #{$value};
	-o-#{$prop}: #{$value};
	#{$prop}: #{$value};
}


	.example{ 
		@include css3-prefix(transition,3s) 
	} 
	.example{ 
		-webkit-transition: 3s;
		-moz-transition: 3s;
		-ms-transition: 3s;
		-o-transition: 3s;
		transition: 3s; 
	}

Media queries for font size

$max-width => maximum ekran ölçüsü , $min-break => minimum ekran ölçüsü , $break => kaç adet font size olacağını belirler 

@mixin font-size-scale($max-width: 1366px, $min-break: 375px, $max: 6) {
	html {
		@for $i from 2 to $max {
			@media(max-width: $max-width - ceil(($max-width - $min-break)/($max - $i))) {
				font-size: 100% - ($i - 1)*6%;
			}
		}
	}
}

Link Colour

$link => button rengi , $visited => :visited rengi , $hover => :focus ve :hover rengi , $active => :active rengi 

@mixin link-colors($link, $visited, $hover, $active) {
	color: $link;
	&:visited {
		color: $visited;
	}
	&:hover,
	&:focus {
		color: $hover;
	}
	&:active {
		color: $active;
	}
}


button{
	@include link-colors(#000,#fff,#b00000,#c5c5c5)
}

button {
	color: #000;
}
  
  button:visited {
	color: #fff;
}
  
  button:hover, button:focus {
	color: #b00000;
}
  
  button:active {
	color: #c5c5c5;
}

Before & After

$display => display (flex) varsayılan block , $content => after veya before content varsayılan ('') 

@mixin before($display: block, $content: '') {
	&:before {
		display: $display;
		content: $content;
		@content;
	}
}

@mixin after($display: block, $content: '') {
	&:after {
		display: $display;
		content: $content;
		@content;
	}
}


.example{
	@include after(inline-block,url(ex.svg)){
		color:red;
	}
	
	@include after(flex,attr(data-id)){
		color:green;
		width:20px;
	}
}

.example:after {
	display: inline-block;
	content: url(ex.svg);
	color: red;
}
  
.example:after {
	display: flex;
	content: attr(data-id);
	color: green;
	width: 20px;
}

Font Set

$name => font name , $url => font url (string olarak girilmesi gerekiyor) , $type => font type (.otf) *font-weight'leri font-set map'in den çekmektedir. 

$font-set: ("light": 200, "regular": 400, "bold": 700, "black":900);
@mixin font-set($name,$url,$type) {
	@each $font-name,
	$font-weight in $font-set {
		@font-face {
			font-family: $name;
			src: url('#{$url}#{$font_weight}.#{$type}') format('opentype');
			font-weight: $font_weight;
			font-style: normal;
		}
	}
}

Font Face

$font-name => font ismi , $file-url => dosya yolu , $file-name => font dosyası ismi , $weight => font-weight varsayılan normal , $style => font-style varsayılan normal 

@mixin font-face($font-name,$file-url,$file-name,$weight: normal,$style: normal) {
	@font-face {
		font-family: quote($font-name);
		src: url($file-url + '/' + $file-name + '.eot');
		src: url($file-url + '/' + $file-name + '.eot?#iefix') format('embedded-opentype'),
		url($file-url + '/' + $file-name + '.woff') format('woff'),
		url($file-url + '/' + $file-name + '.ttf') format('truetype'),
		url($file-url + '/' + $file-name + '.svg##{$font-name}') format('svg');
		font-weight: $weight;
		font-style: $style;
	}
}
